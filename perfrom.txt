import csv
import os
from store.models import Category, Brand, Product

BASE = os.path.join(os.path.dirname(__file__), 'data')

def run():
    # Categories
    with open(os.path.join(BASE, 'categories.csv'), newline='', encoding='utf-8') as f:
        for row in csv.DictReader(f):
            Category.objects.get_or_create(
                name=row["name"],
                slug=row["slug"],
                icon=f'categories/{row["icon"]}'
            )

    # Brands
    with open(os.path.join(BASE, 'brands.csv'), newline='', encoding='utf-8') as f:
        for row in csv.DictReader(f):
            Brand.objects.get_or_create(
                name=row["name"],
                logo=f'brands/{row["logo"]}'
            )

    # Products
    with open(os.path.join(BASE, 'products.csv'), newline='', encoding='utf-8') as f:
        for row in csv.DictReader(f):
            Product.objects.get_or_create(
                title=row["title"],
                category=Category.objects.get(slug=row["category"]),
                brand=Brand.objects.get(name=row["brand"]),
                price=row["price"],
                image=f'products/{row["image"]}',
                is_featured=bool(int(row["is_featured"]))
            )

    print("âœ” Seed Data Loaded Successfully!")

Stores common info (title, brand, base price, etc)

class Product(models.Model):
    name = models.CharField(max_length=255)
    slug = models.SlugField(unique=True)
    brand = models.ForeignKey(Brand, on_delete=models.CASCADE)
    category = models.ForeignKey(Category, on_delete=models.CASCADE)
    description = models.TextField(blank=True)
    gender = models.CharField(max_length=20, choices=[
        ('men', 'Men'),
        ('women', 'Women'),
        ('kids', 'Kids'),
        ('unisex', 'Unisex'),
    ], default='unisex')
    shape = models.CharField(max_length=50, choices=[
        ('round', 'Round'),
        ('rectangle', 'Rectangle'),
        ('square', 'Square'),
        ('oval', 'Oval'),
        ('aviator', 'Aviator'),
        ('cat-eye', 'Cat Eye'),
        ('geometric', 'Geometric'),
    ], blank=True)
    frame_type = models.CharField(max_length=50, choices=[
        ('full-rim', 'Full Rim'),
        ('half-rim', 'Half Rim'),
        ('rimless', 'Rimless'),
    ], blank=True)
    frame_material = models.CharField(max_length=50, blank=True)
    base_price = models.DecimalField(max_digits=10, decimal_places=2)
    is_prescription_supported = models.BooleanField(default=True)
    is_active = models.BooleanField(default=True)
    created_at = models.DateTimeField(auto_now_add=True)

    def __str__(self):
        return self.name

4. Product Variant

Example: Black, Blue, Green color variants

class ProductVariant(models.Model):
    product = models.ForeignKey(Product, on_delete=models.CASCADE, related_name='variants')
    color = models.CharField(max_length=50)
    price = models.DecimalField(max_digits=10, decimal_places=2)
    in_stock = models.PositiveIntegerField(default=0)

    def __str__(self):
        return f"{self.product.name} - {self.color}"

5. Product Images

Multiple images per variant (front, side, 3D etc)

class ProductImage(models.Model):
    variant = models.ForeignKey(ProductVariant, on_delete=models.CASCADE, related_name='images')
    image = models.ImageField(upload_to='products/')
    is_primary = models.BooleanField(default=False)

    def __str__(self):
